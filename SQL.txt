CREATE TABLE

USER_ID VARCHAR2(50) NOT NULL,

USER_NAME VARCHAR2(100) NOT NULL, 
JOB VARCHAR2(3) DEFAULT '00', 
COIN NUMBER(10) DEFAULT 0,
